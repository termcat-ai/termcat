    return window.electron.localFs.read(filePath, maxSizeKB);

    await window.electron.localFs.write(filePath, content);
  }

  async rename(dirPath: string, oldName: string, newName: string): Promise<void> {
    await window.electron.localFs.rename(dirPath, oldName, newName);
  }

    await window.electron.localFs.delete(dirPath, name, isDir);
  }

  async mkdir(dirPath: string, name: string): Promise<void> {
    await window.electron.localFs.mkdir(dirPath, name);
  }

  async createFile(dirPath: string, name: string): Promise<void> {

## 6. Main 进程 — LocalFsHandler

### `src/main/services/local-fs-handler.ts`

```typescript
import * as fs from 'fs';
import * as path from 'path';
import * as os from 'os';
import { exec } from 'child_process';
import { logger, LOG_MODULE } from '../../utils/logger';

const log = logger.withFields({ module: LOG_MODULE.FILE });

/**
 * FileItem 结构（与 Renderer 侧 types/index.ts 一致）
 */
interface FileItem {
  name: string;
  size: string;
  type: string;
  mtime: string;
  permission: string;
  userGroup: string;
  isDir: boolean;
}

/**
 * DirectoryNode 结构
 */
interface DirectoryNode {
  name: string;
  path: string;
  children?: DirectoryNode[];
}

/**
 * 本地文件系统操作（Main 进程）
 *
 * 封装 Node.js fs API，通过 IPC 提供给 Renderer 进程。
 */
export class LocalFsHandler {

  /**
   * 列出目录内容
   */
  async list(dirPath: string): Promise<FileItem[]> {
    const entries = await fs.promises.readdir(dirPath, { withFileTypes: true });
    const items: FileItem[] = [];

    for (const entry of entries) {
      try {
        const fullPath = path.join(dirPath, entry.name);
        const stats = await fs.promises.lstat(fullPath);
        items.push(this.statsToFileItem(entry.name, stats));
      } catch {
        // 无权限访问，跳过
      }
    }

    // 目录排前面，文件排后面
    items.sort((a, b) => {
      if (a.isDir !== b.isDir) return a.isDir ? -1 : 1;
      return a.name.localeCompare(b.name);
    });

    return items;
  }

  /**
   * 获取目录树
   */
  async tree(dirPath: string, maxDepth: number = 3): Promise<DirectoryNode[]> {
    return this.buildTree(dirPath, 0, maxDepth);
  }

  /**
   * 读取文件预览（限制行数）
   */
  async readPreview(filePath: string, maxLines: number = 100): Promise<string> {
    const content = await fs.promises.readFile(filePath, 'utf-8');
    const lines = content.split('\n');
    return lines.slice(0, maxLines).join('\n');
  }

  /**
   * 读取文件完整内容（编辑用，限制大小）
   */
  async read(filePath: string, maxSizeKB: number = 2048): Promise<string> {
    const stats = await fs.promises.stat(filePath);
    if (stats.size > maxSizeKB * 1024) {
      throw new Error(`File too large: ${(stats.size / 1024 / 1024).toFixed(1)}MB (max ${maxSizeKB / 1024}MB)`);
    }
    return fs.promises.readFile(filePath, 'utf-8');
  }

  /**
   * 写入文件
   */
  async write(filePath: string, content: string): Promise<void> {
    await fs.promises.writeFile(filePath, content, 'utf-8');
  }

  /**
   * 重命名
   */
  async rename(dirPath: string, oldName: string, newName: string): Promise<void> {
    const oldPath = path.join(dirPath, oldName);
    const newPath = path.join(dirPath, newName);
    await fs.promises.rename(oldPath, newPath);
  }

  /**
   * 删除
   */
  async delete(dirPath: string, name: string, isDir: boolean): Promise<void> {
    const fullPath = path.join(dirPath, name);
    await fs.promises.rm(fullPath, { recursive: isDir, force: true });
  }

  /**
   * 创建目录
   */
  async mkdir(dirPath: string, name: string): Promise<void> {
    await fs.promises.mkdir(path.join(dirPath, name), { recursive: true });
  }

  /**
   * 创建空文件
   */
  async createFile(dirPath: string, name: string): Promise<void> {
    const fullPath = path.join(dirPath, name);
    await fs.promises.writeFile(fullPath, '', 'utf-8');
  }

  /**
   * 修改权限
   */
  async chmod(dirPath: string, name: string, octal: string): Promise<void> {
    const fullPath = path.join(dirPath, name);
    await fs.promises.chmod(fullPath, parseInt(octal, 8));
  }

  /**
   * 打包文件（tar.gz）
   */
  async pack(dirPath: string, fileNames: string[]): Promise<string> {
    const timestamp = Date.now();
    const tempPath = path.join(os.tmpdir(), `termcat_pack_${timestamp}.tar.gz`);
    const escapedNames = fileNames.map(n => `'${n.replace(/'/g, "'\\''")}'`).join(' ');

    return new Promise<string>((resolve, reject) => {
      exec(
        `cd '${dirPath.replace(/'/g, "'\\''")}' && tar czf '${tempPath}' ${escapedNames}`,
        { timeout: 30000 },
        (error) => {
          if (error) reject(error);
          else resolve(tempPath);
        },
      );
    });
  }

  /**
   * 删除临时文件
   */
  async removeTempFile(tempPath: string): Promise<void> {
    try {
      await fs.promises.rm(tempPath, { force: true });
    } catch {
      // 忽略
    }
  }

  /**
   * 获取 home 目录
   */
  getHomedir(): string {
    return os.homedir();
  }

  // ── Private ──

  private statsToFileItem(name: string, stats: fs.Stats): FileItem {
    const isDir = stats.isDirectory();
    const isLink = stats.isSymbolicLink();
    const isExec = !isDir && (stats.mode & 0o111) !== 0;

    let type = '文件';
    if (isDir) type = '文件夹';
    else if (isLink) type = '链接';
    else if (isExec) type = '可执行文件';

    return {
      name,
      size: isDir ? '-' : this.formatSize(stats.size),
      type,
      mtime: this.formatDate(stats.mtime),
      permission: this.formatPermission(stats.mode, isDir),
      userGroup: `${stats.uid}/${stats.gid}`,
      isDir,
    };
  }

  private formatSize(bytes: number): string {
    if (bytes < 1024) return `${bytes}B`;
    if (bytes < 1024 * 1024) return `${(bytes / 1024).toFixed(1)}K`;
    if (bytes < 1024 * 1024 * 1024) return `${(bytes / 1024 / 1024).toFixed(1)}M`;
    return `${(bytes / 1024 / 1024 / 1024).toFixed(1)}G`;
  }

  private formatDate(date: Date): string {
    const y = date.getFullYear();
    const m = String(date.getMonth() + 1).padStart(2, '0');
    const d = String(date.getDate()).padStart(2, '0');
    const h = String(date.getHours()).padStart(2, '0');
    const min = String(date.getMinutes()).padStart(2, '0');
    return `${y}/${m}/${d} ${h}:${min}`;
  }

  private formatPermission(mode: number, isDir: boolean): string {
    const perms = ['---', '--x', '-w-', '-wx', 'r--', 'r-x', 'rw-', 'rwx'];
    const user = perms[(mode >> 6) & 7];
    const group = perms[(mode >> 3) & 7];
    const other = perms[mode & 7];
    return `${isDir ? 'd' : '-'}${user}${group}${other}`;
  }

  private async buildTree(dirPath: string, depth: number, maxDepth: number): Promise<DirectoryNode[]> {
    if (depth >= maxDepth) return [];

    try {
      const entries = await fs.promises.readdir(dirPath, { withFileTypes: true });
      const nodes: DirectoryNode[] = [];

      for (const entry of entries) {
        if (!entry.isDirectory()) continue;
        if (entry.name.startsWith('.')) continue; // 跳过隐藏目录

        const fullPath = path.join(dirPath, entry.name);
        try {
          const children = await this.buildTree(fullPath, depth + 1, maxDepth);
          nodes.push({
            name: entry.name,
            path: fullPath,
            children,
          });
        } catch {
          // 无权限，跳过
          nodes.push({
            name: entry.name,
            path: fullPath,
            children: [],
          });
        }
      }

      nodes.sort((a, b) => a.name.localeCompare(b.name));
      return nodes;
    } catch {
      return [];
    }
  }
}

export const localFsHandler = new LocalFsHandler();
```

**改动 3**：初始路径同步（约 line 103-124）

```typescript
// 旧：
if (!isInitialPathSyncedRef.current) {
  isInitialPathSyncedRef.current = true;
  window.electron.sshPwd(connectionId)
    .then(pwd => { ... })
    .catch(() => { loadFiles('/'); });
}

// 新：
if (!isInitialPathSyncedRef.current) {
  isInitialPathSyncedRef.current = true;
  if (connectionId) {
    // SSH 模式：从终端获取当前目录
    window.electron.sshPwd(connectionId)
      .then(pwd => {
        if (pwd && pwd.startsWith('/')) loadFiles(pwd);
        else loadFiles('/');
      })
      .catch(() => loadFiles('/'));
  } else {
    // 本地模式：使用 home 目录
    window.electron.localFs.getHomedir()
      .then(home => loadFiles(home))
      .catch(() => loadFiles('/'));
  }
}
```

**改动 5**：隐藏上传/下载按钮（本地模式无意义）

在 `FileListPanel` 或 `FileContextMenu` 中，根据 `connectionId === null` 隐藏：
- 上传按钮
- 下载按钮
- sudo 保存按钮

传递 `isLocal={connectionId === null}` prop 给子组件。

## 11. 实现优先级

| 优先级 | 模块 | 依赖 |
|--------|------|------|
| P0-a | `IFileBrowserService` 接口 | 无 |
| P0-b | `LocalFsHandler`（Main 进程） | 无 |
| P0-c | Preload `localFs` IPC 通道 | 依赖 P0-b |
| P0-d | Main IPC handler 注册 | 依赖 P0-b |
| P1-a | `LocalFileBrowserService` | 依赖 P0-a + P0-c |
| P1-b | `FileBrowserService` implements 接口 | 依赖 P0-a |
| P1-c | `FileBrowserPanel` 路由 + 初始路径 | 依赖 P1-a + P1-b |
| P2 | 隐藏上传/下载/sudo 按钮 | 依赖 P1-c |
