## 2. ICmdExecutor 接口

### `src/services/terminal-backends/ICmdExecutor.ts`

```typescript
/**
 * 一次性命令执行器抽象接口
 *
 * 能力层组件，SSH 和本地各自实现。
 * 由 IHostConnection 持有，上层（SystemMonitorService 等）通过它执行命令。
 *
 * 注意：与 ai-agent 的 ICommandExecutor 不同 —
 * ICommandExecutor 是交互式 Shell 执行器（标记注入、输出解析、超时管理），
 * ICmdExecutor 是简单的一次性命令执行（exec 模式）。
 */

export interface CmdResult {
  output: string;
  exitCode: number;
}

export interface ICmdExecutor {
  /** 执行一条命令，返回输出和退出码 */
  execute(command: string): Promise<CmdResult>;
}
```

## 3. SSHCmdExecutor

### `src/services/terminal-backends/SSHCmdExecutor.ts`

```typescript
import type { ICmdExecutor, CmdResult } from './ICmdExecutor';

/**
 * SSH 命令执行器
 *
 * 通过 sshExecute IPC 在远程服务器执行命令。
 */
export class SSHCmdExecutor implements ICmdExecutor {
  constructor(private connectionId: string) {}

  async execute(command: string): Promise<CmdResult> {
    if (!window.electron) throw new Error('Electron API not available');
    return window.electron.sshExecute(this.connectionId, command);
  }
}
```

## 4. LocalCmdExecutor

### `src/services/terminal-backends/LocalCmdExecutor.ts`

```typescript
import type { ICmdExecutor, CmdResult } from './ICmdExecutor';

/**
 * 本地命令执行器
 *
 * 通过 localExec IPC 在 Main 进程执行本地 Shell 命令。
 */
export class LocalCmdExecutor implements ICmdExecutor {
  async execute(command: string): Promise<CmdResult> {
    if (!window.electron?.localExec) throw new Error('Local exec API not available');
    return window.electron.localExec(command);
  }
}
```

## 7. Main 进程 IPC Handler

### `src/main/main.ts`

```typescript
ipcMain.handle('local-exec', async (_event, command: string) => {
  const { exec } = require('child_process');
  return new Promise<{ output: string; exitCode: number }>((resolve) => {
    exec(command, {
      timeout: 15000,
      maxBuffer: 1024 * 1024,
      shell: process.platform === 'win32' ? 'powershell.exe' : '/bin/sh',
    }, (error: any, stdout: string, stderr: string) => {
      resolve({
        output: stdout || stderr || '',
        exitCode: error ? (error.code || 1) : 0,
      });
    });
  });
});
```

## 8. SystemMonitorService 改动

### `src/plugins/builtin/monitoring-sidebar/services/systemMonitorService.ts`

**改动 1**：构造函数接收 `ICmdExecutor` 替代 `connectionId`

```typescript
// 旧：
constructor(
  connectionId: string,
  updateCallback: (metrics: SystemMetrics) => void,
  hostname?: string,
  osType?: string
)

// 新：
constructor(
  private cmdExecutor: ICmdExecutor,
  updateCallback: (metrics: SystemMetrics) => void,
  hostname?: string,
  osType?: string,
  private isLocal: boolean = false,
)
```

**改动 2**：`fetchSystemMetrics` 中用 cmdExecutor

```typescript
// 旧：
const rttStart = Date.now();
const result = await window.electron.sshExecute(this.connectionId, command);
const rttMs = Date.now() - rttStart;

// 新：
const rttStart = Date.now();
const result = await this.cmdExecutor.execute(command);
const rttMs = this.isLocal ? 0 : (Date.now() - rttStart);
```

**改动 3**：Ping 计算

```typescript
metrics.ping = this.isLocal ? 0 : Math.round(rttMs / 2);
```

## 9. 监控插件 index.ts 改动

### `src/plugins/builtin/monitoring-sidebar/index.ts`

```typescript
context.onConnectionChange((info) => {
  if (monitor) { monitor.stop(); monitor = null; }
  if (!info) return;

  if (info.connectionType === 'local') {
    window.electron.getPlatform().then((platform: string) => {
      const osType = platform === 'darwin' ? 'macos' : platform === 'win32' ? 'windows' : 'linux';
      // 从 HostConnection 获取 cmdExecutor（通过 BottomPanelProps 或直接创建）
      const cmdExecutor = new LocalCmdExecutor();
      monitor = new SystemMonitorService(cmdExecutor, updateCb, 'localhost', osType, true);
      if (info.isVisible && info.isActive) monitor.start(3000);
    });
  } else {
    window.electron.sshGetOSInfo(info.connectionId).then((osInfo) => {
      const cmdExecutor = new SSHCmdExecutor(info.connectionId);
      monitor = new SystemMonitorService(cmdExecutor, updateCb, info.hostname, osInfo?.osType, false);
      if (info.isVisible && info.isActive) monitor.start(3000);
    });
  }
});
```
