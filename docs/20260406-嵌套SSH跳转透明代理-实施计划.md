```typescript
// src/core/terminal/NestedSSHDetector.ts

/**
 * Nested SSH Jump Detector
 *
 * Monitors terminal output stream to detect SSH jumps (user SSHing from host A to host B
 * within the terminal). Maintains a host stack for multi-level jumps.
 * Used by SSHHostConnection to transparently switch fsHandler/cmdExecutor delegates.
 */

import { EventEmitter } from '@/core/ai-agent/EventEmitter';
import { logger, LOG_MODULE } from '@/base/logger/logger';

const log = logger.withFields({ module: LOG_MODULE.TERMINAL });

export interface NestedHost {
  hostname: string;
  username?: string;
  port?: number;
  depth: number;
}

type DetectorState = 'idle' | 'pending_login' | 'nested';

/** Pending SSH attempt not yet confirmed */
interface PendingSSH {
  hostname: string;
  username?: string;
  port?: number;
  timestamp: number;
}

// SSH command pattern: matches "ssh [-p port] [user@]host" from echoed terminal output
// Captures optional user@ prefix and hostname, optional -p port
const SSH_CMD_REGEX = /(?:^|\$\s*|#\s*|>\s*)ssh\s+(?:(?:-[A-Za-z]\s+\S+\s+)*?)(?:-p\s+(\d+)\s+)?(?:([^\s@]+)@)?([^\s;|&]+)/;

// Login success signals
const LOGIN_SUCCESS_PATTERNS = [
  /Last login:/i,
  /Welcome to/i,
  /Linux \S+ \d+\.\d+/,          // Linux kernel banner
  /Ubuntu|Debian|CentOS|Red Hat/i, // OS banner
];

// Login failure signals
const LOGIN_FAIL_PATTERNS = [
  /Connection refused/i,
  /Permission denied/i,
  /No route to host/i,
  /Connection timed out/i,
  /Could not resolve hostname/i,
  /Connection closed by/i,
  /ssh: connect to host/i,
];

// Exit signals (from terminal output, not user input)
const EXIT_PATTERNS = [
  /Connection to \S+ closed/i,
  /logout\s*$/m,
];

// Pending login timeout (ms) — if no success/failure within this time, cancel
const PENDING_TIMEOUT = 30_000;

export class NestedSSHDetector extends EventEmitter {
  private _state: DetectorState = 'idle';
  private _hostStack: NestedHost[] = [];
  private _pending: PendingSSH | null = null;
  private _pendingTimer: ReturnType<typeof setTimeout> | null = null;

  // Line buffer for output stream (terminal output arrives in chunks)
  private _outputBuffer = '';

  get currentHost(): NestedHost | null {
    return this._hostStack.length > 0 ? this._hostStack[this._hostStack.length - 1] : null;
  }

  get isNested(): boolean {
    return this._hostStack.length > 0;
  }

  get depth(): number {
    return this._hostStack.length;
  }

  get hostStack(): readonly NestedHost[] {
    return this._hostStack;
  }

  /**
   * Feed terminal output data for analysis.
   * Called by SSHHostConnection from terminal.onData().
   */
  feedOutput(data: string): void {
    this._outputBuffer += data;

    // Process complete lines (keep partial last line in buffer)
    const lines = this._outputBuffer.split(/\r?\n/);
    // Keep the last (possibly incomplete) line in buffer
    this._outputBuffer = lines.pop() || '';

    for (const line of lines) {
      this._processLine(this._stripAnsi(line));
    }

    // Also check the current partial line for certain patterns
    // (some signals like "Connection to X closed" may not end with newline immediately)
    if (this._outputBuffer.length > 0) {
      this._checkPartialLine(this._stripAnsi(this._outputBuffer));
    }
  }

  dispose(): void {
    this._clearPendingTimer();
    this._hostStack = [];
    this._state = 'idle';
    this._pending = null;
    this._outputBuffer = '';
    this.removeAllListeners();
  }

  // ── Private ──

  private _processLine(line: string): void {
    const trimmed = line.trim();
    if (!trimmed) return;

    switch (this._state) {
      case 'idle':
        this._detectSSHCommand(trimmed);
        break;
      case 'pending_login':
        this._checkLoginResult(trimmed);
        break;
      case 'nested':
        // In nested state: check for new SSH command (multi-level) or exit
        this._detectSSHCommand(trimmed);
        this._detectExit(trimmed);
        break;
    }
  }

  private _checkPartialLine(line: string): void {
    const trimmed = line.trim();
    if (!trimmed) return;

    if (this._state === 'pending_login') {
      this._checkLoginResult(trimmed);
    } else if (this._state === 'nested') {
      this._detectExit(trimmed);
    }
  }

  private _detectSSHCommand(line: string): void {
    const match = line.match(SSH_CMD_REGEX);
    if (!match) return;

    const port = match[1] ? parseInt(match[1], 10) : undefined;
    const username = match[2] || undefined;
    const hostname = match[3];

    // Ignore if hostname looks like a flag or option
    if (hostname.startsWith('-')) return;

    log.info('nested-ssh.detected_cmd', 'SSH command detected in output', {
      hostname, username, port,
    });

    this._pending = { hostname, username, port, timestamp: Date.now() };
    this._state = 'pending_login';

    // Start timeout — if login doesn't succeed/fail in time, cancel
    this._clearPendingTimer();
    this._pendingTimer = setTimeout(() => {
      if (this._state === 'pending_login') {
        log.info('nested-ssh.pending_timeout', 'Pending SSH login timed out', {
          hostname: this._pending?.hostname,
        });
        this._pending = null;
        this._state = this._hostStack.length > 0 ? 'nested' : 'idle';
      }
    }, PENDING_TIMEOUT);
  }

  private _checkLoginResult(line: string): void {
    // Check for failure first
    for (const pattern of LOGIN_FAIL_PATTERNS) {
      if (pattern.test(line)) {
        log.info('nested-ssh.login_failed', 'SSH login failed', {
          hostname: this._pending?.hostname, signal: line.substring(0, 80),
        });
        this._clearPendingTimer();
        this._pending = null;
        this._state = this._hostStack.length > 0 ? 'nested' : 'idle';
        return;
      }
    }

    // Check for success
    for (const pattern of LOGIN_SUCCESS_PATTERNS) {
      if (pattern.test(line)) {
        this._confirmLogin();
        return;
      }
    }
  }

  private _confirmLogin(): void {
    if (!this._pending) return;

    this._clearPendingTimer();

    const nestedHost: NestedHost = {
      hostname: this._pending.hostname,
      username: this._pending.username,
      port: this._pending.port,
      depth: this._hostStack.length + 1,
    };

    this._hostStack.push(nestedHost);
    this._state = 'nested';
    this._pending = null;

    log.info('nested-ssh.entered', 'Nested SSH confirmed', {
      hostname: nestedHost.hostname, depth: nestedHost.depth,
    });

    this.emit('host-entered', nestedHost, [...this._hostStack]);
  }

  private _detectExit(line: string): void {
    for (const pattern of EXIT_PATTERNS) {
      if (pattern.test(line)) {
        this._confirmExit(line);
        return;
      }
    }
  }

  private _confirmExit(signal: string): void {
    if (this._hostStack.length === 0) return;

    const exited = this._hostStack.pop()!;
    const current = this._hostStack.length > 0
      ? this._hostStack[this._hostStack.length - 1]
      : null;

    this._state = this._hostStack.length > 0 ? 'nested' : 'idle';

    log.info('nested-ssh.exited', 'Returned from nested SSH', {
      exited_host: exited.hostname,
      returned_to: current?.hostname ?? '(original)',
      signal: signal.substring(0, 80),
    });

    this.emit('host-exited', current, [...this._hostStack]);
  }

  private _clearPendingTimer(): void {
    if (this._pendingTimer) {
      clearTimeout(this._pendingTimer);
      this._pendingTimer = null;
    }
  }

  /** Strip ANSI escape sequences from terminal output */
  private _stripAnsi(str: string): string {
    // eslint-disable-next-line no-control-regex
    return str.replace(/\x1b\[[0-9;]*[A-Za-z]/g, '')
              .replace(/\x1b\][^\x07]*\x07/g, '')   // OSC sequences
              .replace(/\x1b[()][0-9A-Za-z]/g, '');  // Character set sequences
  }
}
```

```typescript
// src/core/terminal/TerminalCmdExecutor.ts

/**
 * Terminal-based Command Executor
 *
 * Executes commands by writing to terminal stdin and parsing stdout.
 * Uses unique markers to delimit command output from the surrounding terminal stream.
 * Used when nested SSH makes the original SSH exec channel unreachable (e.g., user
 * SSHed from Host A to Host B in the terminal — Host B has no dedicated exec channel).
 *
 * Limitations:
 * - Serial execution only (one command at a time)
 * - Marker strings appear briefly in terminal output
 * - Not suitable for interactive commands (vim, top, etc.)
 */

import type { ICmdExecutor, CmdResult } from './ICmdExecutor';
import type { ITerminalBackend } from './ITerminalBackend';
import { logger, LOG_MODULE } from '@/base/logger/logger';

const log = logger.withFields({ module: LOG_MODULE.TERMINAL });

/** Default command timeout (ms) */
const DEFAULT_TIMEOUT = 30_000;

/** Marker prefix — intentionally ugly to avoid collision with normal output */
const MARKER_PREFIX = '___TERMCAT_CMD_';

export class TerminalCmdExecutor implements ICmdExecutor {
  private _terminal: ITerminalBackend;
  private _queue: Array<() => void> = [];
  private _running = false;
  private _dataHandler: ((data: string) => void) | null = null;
  private _timeout: number;

  constructor(terminal: ITerminalBackend, timeout: number = DEFAULT_TIMEOUT) {
    this._terminal = terminal;
    this._timeout = timeout;
  }

  async execute(command: string): Promise<CmdResult> {
    // Serialize: wait for previous command to finish
    if (this._running) {
      await new Promise<void>(resolve => this._queue.push(resolve));
    }
    this._running = true;

    try {
      return await this._executeInternal(command);
    } finally {
      this._running = false;
      // Kick next queued command
      const next = this._queue.shift();
      if (next) next();
    }
  }

  private _executeInternal(command: string): Promise<CmdResult> {
    return new Promise<CmdResult>((resolve, reject) => {
      const id = Math.random().toString(36).substring(2, 10);
      const startMarker = `${MARKER_PREFIX}START_${id}`;
      const exitMarker = `${MARKER_PREFIX}EXIT_${id}_`;
      const endMarker = `${MARKER_PREFIX}END_${id}`;

      let outputBuffer = '';
      let resolved = false;

      // Timeout guard
      const timer = setTimeout(() => {
        if (resolved) return;
        resolved = true;
        this._removeDataHandler();
        log.warn('terminal-cmd.timeout', 'Command execution timed out', {
          command: command.substring(0, 100), timeout: this._timeout,
        });
        resolve({ output: outputBuffer, exitCode: -1 });
      }, this._timeout);

      // Listen to terminal output
      this._dataHandler = (data: string) => {
        if (resolved) return;
        outputBuffer += data;

        // Check if end marker has arrived
        const endIdx = outputBuffer.indexOf(endMarker);
        if (endIdx === -1) return;

        resolved = true;
        clearTimeout(timer);
        this._removeDataHandler();

        // Parse output between markers
        const result = this._parseOutput(outputBuffer, startMarker, exitMarker, endMarker);
        resolve(result);
      };

      this._terminal.onData(this._dataHandler);

      // Build wrapped command:
      // echo 'START_MARKER'; { command; } 2>&1; echo "EXIT_MARKER$?"; echo 'END_MARKER'
      const wrappedCmd = [
        `echo '${startMarker}'`,
        `{ ${command}; } 2>&1`,
        `echo "${exitMarker}$?"`,
        `echo '${endMarker}'`,
      ].join('; ');

      this._terminal.write(wrappedCmd + '\n');
    });
  }

  private _parseOutput(
    raw: string,
    startMarker: string,
    exitMarker: string,
    endMarker: string,
  ): CmdResult {
    try {
      // Find start marker
      const startIdx = raw.indexOf(startMarker);
      const endIdx = raw.indexOf(endMarker);

      if (startIdx === -1 || endIdx === -1) {
        return { output: '', exitCode: -1 };
      }

      // Content between start marker line end and exit marker
      const afterStart = raw.indexOf('\n', startIdx);
      const contentRaw = raw.substring(
        afterStart !== -1 ? afterStart + 1 : startIdx + startMarker.length,
        endIdx,
      );

      // Extract exit code from exitMarker line
      let exitCode = 0;
      const exitIdx = contentRaw.lastIndexOf(exitMarker);
      let output = contentRaw;

      if (exitIdx !== -1) {
        const exitLine = contentRaw.substring(exitIdx + exitMarker.length).trim();
        const exitLineEnd = exitLine.indexOf('\n');
        const codeStr = exitLineEnd !== -1 ? exitLine.substring(0, exitLineEnd) : exitLine;
        exitCode = parseInt(codeStr, 10) || 0;
        output = contentRaw.substring(0, exitIdx);
      }

      // Clean up: remove trailing newlines and the echoed command itself
      output = output.replace(/^\s*\n/, '').replace(/\n\s*$/, '');

      return { output, exitCode };
    } catch (error) {
      log.error('terminal-cmd.parse_failed', 'Failed to parse command output', {
        error: error instanceof Error ? error.message : 'Unknown',
      });
      return { output: '', exitCode: -1 };
    }
  }

  /**
   * Remove the data handler from terminal.
   * Note: ITerminalBackend.onData() adds to a callback array but has no removeListener.
   * We null out the handler reference so subsequent calls are no-ops.
   * The callback remains in the array but is lightweight (just a null check + return).
   */
  private _removeDataHandler(): void {
    this._dataHandler = null;
  }
}
```

```typescript
// src/core/terminal/ProxyCmdExecutor.ts

/**
 * Transparent ICmdExecutor Proxy
 *
 * Routes command execution to either the original SSH executor or a terminal-based
 * executor depending on nested SSH state. Upper layers call execute() without
 * knowing which backend is active.
 */

import type { ICmdExecutor, CmdResult } from './ICmdExecutor';
import type { TerminalCmdExecutor } from './TerminalCmdExecutor';

export class ProxyCmdExecutor implements ICmdExecutor {
  private _original: ICmdExecutor;
  private _nested: TerminalCmdExecutor | null = null;
  private _isNested = false;

  constructor(original: ICmdExecutor) {
    this._original = original;
  }

  async execute(command: string): Promise<CmdResult> {
    if (this._isNested && this._nested) {
      return this._nested.execute(command);
    }
    return this._original.execute(command);
  }

  /** Switch to nested mode (internal, called by SSHHostConnection) */
  _switchToNested(termExecutor: TerminalCmdExecutor): void {
    this._nested = termExecutor;
    this._isNested = true;
  }

  /** Restore original mode (internal, called by SSHHostConnection) */
  _switchToOriginal(): void {
    this._isNested = false;
    this._nested = null;
  }

  /** Whether currently in nested mode */
  get isNested(): boolean {
    return this._isNested;
  }
}
```

该类通过 `TerminalCmdExecutor` 执行 shell 命令来实现 `IFsHandler` 接口。解析逻辑复用 `SSHFsHandler` 中已验证的 `ls -alh` 解析模式。

```typescript
// src/core/terminal/TerminalFsHandler.ts

/**
 * Terminal-based File System Handler
 *
 * Implements IFsHandler by executing shell commands through TerminalCmdExecutor.
 * Used when nested SSH makes the original SFTP channel unreachable.
 *
 * Limitations:
 * - downloadFile/uploadFile not supported (no direct file transfer channel)
 * - Performance is lower than SFTP due to command serialization
 * - Large file operations limited by base64 encoding overhead
 */

import type { IFsHandler, DirectoryNode } from './IFsHandler';
import type { ICmdExecutor } from './ICmdExecutor';
import { FileItem } from '@/utils/types';
import { logger, LOG_MODULE } from '@/base/logger/logger';

const log = logger.withFields({ module: LOG_MODULE.FILE });

const TRANSFER_NOT_SUPPORTED_MSG =
  'File transfer is not supported in nested SSH mode. Please use scp or configure a direct connection.';

export class TerminalFsHandler implements IFsHandler {
  constructor(private executor: ICmdExecutor) {}

  async listFiles(path: string): Promise<FileItem[]> {
    const command = `ls -alh --time-style="+%Y/%m/%d %H:%M" "${path}" 2>/dev/null || ls -alh "${path}"`;
    const result = await this.executor.execute(command);
    if (!result.output) return [];
    return this._parseFileList(result.output);
  }

  async getDirectoryTree(path: string = '/', maxDepth: number = 3): Promise<DirectoryNode[]> {
    const command = `find "${path}" -maxdepth ${maxDepth} -type d 2>/dev/null | sort`;
    const result = await this.executor.execute(command);
    if (!result.output) return [];
    return this._buildDirectoryTree(result.output, path);
  }

  async getFileContent(filePath: string, maxLines: number = 100): Promise<string> {
    const result = await this.executor.execute(`head -n ${maxLines} "${filePath}"`);
    return result.output || '';
  }

  async readFileForEdit(filePath: string, maxSizeKB: number = 2048): Promise<string> {
    // Check file size
    const sizeResult = await this.executor.execute(
      `stat -c%s "${filePath}" 2>/dev/null || stat -f%z "${filePath}" 2>/dev/null`
    );
    const fileSize = parseInt(sizeResult.output?.trim() || '0', 10);
    if (fileSize > maxSizeKB * 1024) {
      throw new Error(`File too large: ${(fileSize / 1024).toFixed(0)}KB (max ${maxSizeKB}KB)`);
    }
    // Read with base64 encoding
    const result = await this.executor.execute(`cat "${filePath}" | base64`);
    if (result.exitCode !== 0) throw new Error(result.output || 'Failed to read file');
    const base64Str = (result.output || '').replace(/\s/g, '');
    if (!base64Str) return '';
    const binaryStr = atob(base64Str);
    const bytes = new Uint8Array(binaryStr.length);
    for (let i = 0; i < binaryStr.length; i++) {
      bytes[i] = binaryStr.charCodeAt(i);
    }
    return new TextDecoder('utf-8').decode(bytes);
  }

  async writeFileContent(filePath: string, content: string): Promise<void> {
    const encoder = new TextEncoder();
    const bytes = encoder.encode(content);
    let binary = '';
    for (let i = 0; i < bytes.length; i++) {
      binary += String.fromCharCode(bytes[i]);
    }
    const base64Str = btoa(binary);
    const chunkSize = 65536;
    if (base64Str.length <= chunkSize) {
      const result = await this.executor.execute(
        `echo '${base64Str}' | base64 -d > "${filePath}"`
      );
      if (result.exitCode !== 0) throw new Error(result.output || 'Write failed');
    } else {
      const tmpPath = `/tmp/termcat_edit_${Date.now()}.b64`;
      for (let i = 0; i < base64Str.length; i += chunkSize) {
        const chunk = base64Str.slice(i, i + chunkSize);
        const op = i === 0 ? '>' : '>>';
        const result = await this.executor.execute(
          `printf '%s' '${chunk}' ${op} "${tmpPath}"`
        );
        if (result.exitCode !== 0) throw new Error(result.output || 'Write chunk failed');
      }
      const decodeResult = await this.executor.execute(
        `base64 -d < "${tmpPath}" > "${filePath}" && rm -f "${tmpPath}"`
      );
      if (decodeResult.exitCode !== 0) throw new Error(decodeResult.output || 'Decode failed');
    }
    log.info('terminal-fs.write.success', 'File written via terminal', { filePath, size: content.length });
  }

  async writeFileContentSudo(filePath: string, content: string, password: string): Promise<void> {
    const encoder = new TextEncoder();
    const bytes = encoder.encode(content);
    let binary = '';
    for (let i = 0; i < bytes.length; i++) {
      binary += String.fromCharCode(bytes[i]);
    }
    const base64Str = btoa(binary);
    const result = await this.executor.execute(
      `echo '${password}' | sudo -S sh -c "echo '${base64Str}' | base64 -d > '${filePath}'" 2>&1`
    );
    if (result.exitCode !== 0) {
      const output = result.output || '';
      if (output.includes('incorrect password') || output.includes('Sorry, try again')) {
        throw new Error('Incorrect sudo password');
      }
      throw new Error(output || 'Sudo write failed');
    }
  }

  async rename(dirPath: string, oldName: string, newName: string): Promise<void> {
    const oldPath = dirPath === '/' ? `/${oldName}` : `${dirPath}/${oldName}`;
    const newPath = dirPath === '/' ? `/${newName}` : `${dirPath}/${newName}`;
    const result = await this.executor.execute(`mv "${oldPath}" "${newPath}"`);
    if (result.exitCode !== 0) throw new Error(result.output || 'Rename failed');
  }

  async deleteFile(dirPath: string, name: string, isDir: boolean): Promise<void> {
    const fullPath = dirPath === '/' ? `/${name}` : `${dirPath}/${name}`;
    const cmd = isDir ? `rm -rf "${fullPath}"` : `rm -f "${fullPath}"`;
    const result = await this.executor.execute(cmd);
    if (result.exitCode !== 0) throw new Error(result.output || 'Delete failed');
  }

  async mkdir(dirPath: string, name: string): Promise<void> {
    const fullPath = dirPath === '/' ? `/${name}` : `${dirPath}/${name}`;
    const result = await this.executor.execute(`mkdir -p "${fullPath}"`);

  private _parseFileList(output: string): FileItem[] {
    const files: FileItem[] = [];
    const lines = output.trim().split('\n');
    // Skip first line (total xxx)
    for (let i = 1; i < lines.length; i++) {
      const line = lines[i].trim();
      if (!line) continue;
      try {
        const file = this._parseFileLine(line);
        if (file && file.name !== '.' && file.name !== '..') {
          files.push(file);
        }
      } catch {
        // Skip unparseable lines
      }
    }
    return files;
  }

  /**
   * Parse single line of file info
   * Linux --time-style: drwxr-xr-x 2 user group 4.0K 2026/01/16 11:23 filename
   * macOS/BSD default:  drwxr-xr-x 2 user group 4.0K Jan 16 00:39 filename
   */
  private _parseFileLine(line: string): FileItem | null {
    const parts = line.split(/\s+/);
    if (parts.length < 8) return null;

    const permission = parts[0];
    if (!/^[d\-lc][rwx\-sStT]{9}/.test(permission)) return null;

    const user = parts[2];
    const group = parts[3];
    const size = parts[4];

    let mtime: string;
    let nameIndex: number;
    if (/^\d{4}\//.test(parts[5])) {
      mtime = `${parts[5]} ${parts[6]}`;
      nameIndex = 7;
    } else {
      mtime = `${parts[5]} ${parts[6]} ${parts[7]}`;
      nameIndex = 8;
    }

    const name = parts.slice(nameIndex).join(' ');
    const isDir = permission.startsWith('d');
    const isSymlink = permission.startsWith('l');

    let fileName = name;
    if (isSymlink && name.includes(' -> ')) {
      fileName = name.split(' -> ')[0];
    }

    let type = 'File';
    if (isDir) type = 'Directory';
    else if (isSymlink) type = 'Link';
    else if (permission.includes('x')) type = 'Executable';

    return {
      name: fileName,
      size: isDir ? '-' : size,
      type,
      mtime,
      permission,
      userGroup: `${user}/${group}`,
      isDir,
    };
  }

  private _buildDirectoryTree(output: string, rootPath: string): DirectoryNode[] {
    const paths = output.trim().split('\n').filter(p => p);
    const root: DirectoryNode = { name: rootPath, path: rootPath, children: [] };
    const pathMap = new Map<string, DirectoryNode>();
    pathMap.set(rootPath, root);

    for (const fullPath of paths) {
      if (fullPath === rootPath) continue;
      const parts = fullPath.split('/').filter(p => p);
      let currentPath = '';

      for (let i = 0; i < parts.length; i++) {
        const part = parts[i];
        const parentPath = currentPath || '/';
        currentPath = currentPath ? `${currentPath}/${part}` : `/${part}`;

        if (!pathMap.has(currentPath)) {
          const node: DirectoryNode = { name: part, path: currentPath, children: [] };
          pathMap.set(currentPath, node);
          const parent = pathMap.get(parentPath);
          if (parent) {
            parent.children = parent.children || [];
            parent.children.push(node);
          }
        }
      }
    }

    return root.children || [];
  }
}
```

```typescript
export class SSHHostConnection implements IHostConnection {
  readonly type: HostConnectionType = 'ssh';

  private _id: string = '';
  private _terminal: SSHTerminalBackend | null = null;
  private _proxyFs: ProxyFsHandler | null = null;
  private _proxyCmd: ProxyCmdExecutor | null = null;
  private _detector: NestedSSHDetector | null = null;
  private _effectiveHostname: string | null = null;
  private _hostChangedCallbacks: Array<(hostname: string) => void> = [];
  private _isConnected = false;

  constructor(private host: Host) {}

  get id(): string { return this._id; }

  get terminal(): SSHTerminalBackend | null {
    return this._terminal;
  }

  get fsHandler(): IFsHandler | null {
    return this._proxyFs;
  }

  get cmdExecutor(): ICmdExecutor | null {
    return this._proxyCmd;
  }

  get effectiveHostname(): string | null {
    return this._effectiveHostname ?? this.host.hostname;
  }

  onHostChanged(cb: (hostname: string) => void): () => void {
    this._hostChangedCallbacks.push(cb);
    return () => {
      this._hostChangedCallbacks = this._hostChangedCallbacks.filter(c => c !== cb);
    };
  }

  async connect(): Promise<void> {
    log.info('ssh-host.connecting', 'SSHHostConnection connecting', {
      host_id: this.host.id, hostname: this.host.hostname,
    });

    const session = await sshService.connect(this.host);

    if (!session.connectionId) {
      throw new Error('SSH connection failed: no connectionId');
    }

    this._id = session.connectionId;
    this._terminal = new SSHTerminalBackend(
      session.connectionId,
      this.host.terminal?.encoding,
    );

    // Wrap original handlers in proxies
    const sshFs = new SSHFsHandler(session.connectionId);
    const sshCmd = new SSHCmdExecutor(session.connectionId);
    this._proxyFs = new ProxyFsHandler(sshFs);
    this._proxyCmd = new ProxyCmdExecutor(sshCmd);

    // Initialize nested SSH detector
    this._detector = new NestedSSHDetector();
    this._setupNestedDetection();

    this._isConnected = true;

    log.info('ssh-host.connected', 'SSHHostConnection connected', {
      connection_id: this._id,
    });
  }

  private _setupNestedDetection(): void {
    if (!this._detector || !this._terminal) return;

    // Feed terminal output to detector
    this._terminal.onData(data => this._detector?.feedOutput(data));

    // Handle entering nested host
    this._detector.on('host-entered', (nestedHost: NestedHost) => {
      const termCmd = new TerminalCmdExecutor(this._terminal!);
      const termFs = new TerminalFsHandler(termCmd);
      this._proxyCmd!._switchToNested(termCmd);
      this._proxyFs!._switchToNested(termFs);
      this._effectiveHostname = nestedHost.hostname;

      for (const cb of this._hostChangedCallbacks) {
        try { cb(nestedHost.hostname); } catch { /* ignore */ }
      }

      log.info('nested-ssh.proxy_switched', 'Proxy switched to nested host', {
        hostname: nestedHost.hostname, depth: nestedHost.depth,
      });
    });

    // Handle exiting nested host
    this._detector.on('host-exited', (current: NestedHost | null) => {
      if (current) {
        // Returned to a higher-level nested host — rebuild terminal executor
        const termCmd = new TerminalCmdExecutor(this._terminal!);
        const termFs = new TerminalFsHandler(termCmd);
        this._proxyCmd!._switchToNested(termCmd);
        this._proxyFs!._switchToNested(termFs);
        this._effectiveHostname = current.hostname;
      } else {
        // Returned to original host
        this._proxyCmd!._switchToOriginal();
        this._proxyFs!._switchToOriginal();
        this._effectiveHostname = null;
      }

      const hostname = current?.hostname ?? this.host.hostname;
      for (const cb of this._hostChangedCallbacks) {
        try { cb(hostname); } catch { /* ignore */ }
      }

      log.info('nested-ssh.proxy_restored', 'Proxy restored', {
        hostname, depth: current?.depth ?? 0,
      });
    });
  }

  async disconnect(): Promise<void> {
    if (this._id) {
      sshService.disconnectSession(this._id);
    }
    this._isConnected = false;
  }

  dispose(): void {
    this._detector?.dispose();
    this._terminal?.dispose();
    if (this._isConnected) {
      this.disconnect();
    }
    this._terminal = null;
    this._detector = null;
    this._hostChangedCallbacks = [];
  }
}
```
