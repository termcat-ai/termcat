## 4. 抽象终端接口

### 4.1 `src/services/terminal-backends/ITerminalBackend.ts`

```typescript
import { TerminalBackendType, TerminalConnectOptions } from '@/types/terminal';

/**
 * 数据回调函数类型
 */
export type DataCallback = (data: string) => void;

/**
 * 关闭回调函数类型
 */
export type CloseCallback = () => void;

/**
 * 抽象终端后端接口
 *
 * SSH 终端和本地终端各自独立实现此接口，
 * 上层组件（XTermTerminal 等）仅依赖此接口，不感知底层实现。
 */
export interface ITerminalBackend {
  /** 后端类型标识 */
  readonly type: TerminalBackendType;

  /** 连接/进程标识符（connectionId 或 ptyId） */
  readonly id: string;

  /** 当前是否已连接 */
  readonly isConnected: boolean;

  /**
   * 建立连接
   * - SSH: 发起 SSH 连接 + 创建 Shell
   * - Local: 创建 PTY 进程
   */
  connect(options: TerminalConnectOptions): Promise<void>;

  /**
   * 断开连接
   * - SSH: 断开 SSH 连接
   * - Local: 销毁 PTY 进程
   */
  disconnect(): Promise<void>;

  /**
   * 写入用户输入
   */
  write(data: string): void;

  /**
   * 调整终端大小
   */
  resize(cols: number, rows: number): void;

  /**
   * 注册数据输出回调（终端输出）
   */
  onData(callback: DataCallback): void;

  /**
   * 注册连接关闭回调
   */
  onClose(callback: CloseCallback): void;

  /**
   * 释放所有资源和事件监听器
   * 组件卸载时调用
   */
  dispose(): void;
}
```

## 8. Main 进程 — local-pty-service.ts

### 8.1 类结构

```typescript
import * as pty from 'node-pty';

/**
 * PTY 实例信息
 */
interface PtyInstance {
  id: string;
  process: pty.IPty;
  shell: string;
  cwd: string;
  webContents: Electron.WebContents;
  createdAt: number;
}

/**
 * 本地 PTY 服务
 *
 * Main 进程中管理本地终端进程，与 ssh-service.ts 平行独立。
 */
export class LocalPtyService {
  private instances: Map<string, PtyInstance>;

  /**
   * 检测当前平台可用的 Shell 列表
   * - macOS/Linux: 读取 /etc/shells
   * - Windows: 探测已知 Shell 路径
   * 返回 ShellInfo[]
   */
  async detectShells(): Promise<ShellInfo[]>;

  /**
   * 获取当前平台的默认 Shell
   * - macOS/Linux: process.env.SHELL 或 /bin/sh
   * - Windows: powershell.exe
   */
  getDefaultShell(): ShellInfo;

  /**
   * 创建本地 PTY 进程
   * 调用 pty.spawn() 并注册数据监听
   * 返回 ptyId
   */
  create(options: {
    shell?: string;
    args?: string[];
    cwd?: string;
    env?: Record<string, string>;
    cols: number;
    rows: number;
    webContents: Electron.WebContents;
  }): string;

  /**
   * 写入数据到 PTY
   */
  write(ptyId: string, data: string): boolean;

  /**
   * 调整 PTY 大小
   */
  resize(ptyId: string, cols: number, rows: number): boolean;

  /**
   * 销毁 PTY 进程
   * kill 进程 + 从 Map 移除 + 通知 Renderer
   */
  destroy(ptyId: string): void;

  /**
   * 销毁所有 PTY 进程
   * 应用退出时调用
   */
  destroyAll(): void;

  /**
   * 检查 PTY 是否存在
   */
  exists(ptyId: string): boolean;
}
```

### 8.3 PTY 创建核心逻辑

```typescript
create(options): string {
  const ptyId = `pty-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;

  const shell = options.shell || this.getDefaultShell().path;
  const args = options.args || [];

  const ptyProcess = pty.spawn(shell, args, {
    name: 'xterm-256color',
    cols: options.cols,
    rows: options.rows,
    cwd: options.cwd || os.homedir(),
    env: { ...process.env, ...options.env },
    // Windows 特有：使用 ConPTY
    useConpty: process.platform === 'win32',
  });

  // 数据输出 → IPC event → Renderer
  ptyProcess.onData((data) => {
    options.webContents.send('local-pty-data', ptyId, data);
  });

  // 进程退出 → 清理 + 通知
  ptyProcess.onExit(({ exitCode, signal }) => {
    options.webContents.send('local-pty-close', ptyId, exitCode);
    this.instances.delete(ptyId);
  });

  this.instances.set(ptyId, {
    id: ptyId,
    process: ptyProcess,
    shell,
    cwd: options.cwd || os.homedir(),
    webContents: options.webContents,
    createdAt: Date.now(),
  });

  return ptyId;
}
```

localTerminal: {
  // invoke 类型（请求-响应）
  create: (options: {
    shell?: string;
    args?: string[];
    cwd?: string;
    env?: Record<string, string>;
    cols: number;
    rows: number;
  }) => ipcRenderer.invoke('local-pty-create', options),

  destroy: (ptyId: string) =>
    ipcRenderer.invoke('local-pty-destroy', ptyId),

  resize: (ptyId: string, cols: number, rows: number) =>
    ipcRenderer.invoke('local-pty-resize', ptyId, cols, rows),

  getShells: () =>
    ipcRenderer.invoke('local-pty-get-shells'),

  getDefaultShell: () =>
    ipcRenderer.invoke('local-pty-get-default-shell'),

  // send 类型（单向）
  write: (ptyId: string, data: string) =>
    ipcRenderer.send('local-pty-write', ptyId, data),

  // event 类型（Main → Renderer）
  onData: (callback: (ptyId: string, data: string) => void) => {
    const handler = (_event: any, ptyId: string, data: string) =>
      callback(ptyId, data);
    ipcRenderer.on('local-pty-data', handler);
    return () => ipcRenderer.removeListener('local-pty-data', handler);
  },

  onClose: (callback: (ptyId: string, exitCode: number) => void) => {
    const handler = (_event: any, ptyId: string, exitCode: number) =>
      callback(ptyId, exitCode);
    ipcRenderer.on('local-pty-close', handler);
    return () => ipcRenderer.removeListener('local-pty-close', handler);
  },
},

## 10. Main 进程 IPC Handler 注册

### 10.1 `src/main/main.ts` 新增部分

```typescript
import { LocalPtyService } from './local-pty-service';

const localPtyService = new LocalPtyService();

// 创建本地 PTY
ipcMain.handle('local-pty-create', async (event, options) => {
  const ptyId = localPtyService.create({
    ...options,
    webContents: event.sender,
  });
  return { ptyId };
});

// 销毁 PTY
ipcMain.handle('local-pty-destroy', async (event, ptyId) => {
  localPtyService.destroy(ptyId);
  return { success: true };
});

// 调整大小
ipcMain.handle('local-pty-resize', async (event, ptyId, cols, rows) => {
  return { success: localPtyService.resize(ptyId, cols, rows) };
});

// 获取可用 Shell 列表
ipcMain.handle('local-pty-get-shells', async () => {
  return localPtyService.detectShells();
});

// 获取默认 Shell
ipcMain.handle('local-pty-get-default-shell', async () => {
  return localPtyService.getDefaultShell();
});

// 写入数据（send 类型，无返回值）
ipcMain.on('local-pty-write', (event, ptyId, data) => {
  localPtyService.write(ptyId, data);
});

// 应用退出时清理
app.on('before-quit', () => {
  localPtyService.destroyAll();
});

  <Radio value="direct">{t.hostConfig.sshDirect}</Radio>
  <Radio value="jump">{t.hostConfig.sshJump}</Radio>
  <Radio value="local">{t.hostConfig.localTerminal}</Radio>
</RadioGroup>

// 根据 connectionType 条件渲染不同配置区域
{connectionType === 'local' ? (
  <LocalTerminalConfigSection />
) : (
  <SSHConfigSection />  // 现有 SSH 配置
)}
```
