## 3. IHostConnection — host 连接统一入口

### 3.1 `src/services/terminal-backends/IHostConnection.ts`

```typescript
import { ITerminalBackend } from './ITerminalBackend';
import { ICommandExecutor } from '@/modules/ai-agent/ICommandExecutor';

/**
 * 文件传输接口（SSH 专属能力）
 */
export interface IFileTransfer {
  upload(localPath: string, remotePath: string): Promise<void>;
  download(remotePath: string, localPath: string): Promise<void>;
  uploadDirectory(localPath: string, remotePath: string): Promise<void>;
  downloadDirectory(remotePath: string, localPath: string): Promise<void>;
}

/**
 * Host 连接统一入口
 *
 * 能力层的组合接口，聚合终端 I/O、命令执行、文件传输。
 * SSH 和本地各自实现，上层（TerminalView、插件）仅持有此接口。
 */
export interface IHostConnection {
  /** 连接类型 */
  readonly type: 'ssh' | 'local';

  /** 连接生命周期 */
  connect(): Promise<void>;
  disconnect(): Promise<void>;

  /** 终端 I/O（xterm.js 数据流） */
  readonly terminal: ITerminalBackend;

  /** 命令执行（AI Agent 使用） */
  readonly executor: ICommandExecutor;

  /** 文件传输（SSH 专属，本地为 null） */
  readonly transfer: IFileTransfer | null;

  /** 释放所有资源 */
  dispose(): void;
}
```

## 5. AI Agent 本地执行器

### 4.1 `src/modules/ai-agent/executors/LocalShellExecutor.ts`

```typescript
import { BaseShellExecutor } from './BaseShellExecutor';

/**
 * 本地 Shell 命令执行器
 *
 * 继承 BaseShellExecutor，通过 Local PTY IPC 执行命令。
 * 复用基类所有通用逻辑（标记注入、输出解析、超时管理等）。
 */
export class LocalShellExecutor extends BaseShellExecutor {
  private ptyId: string = '';
  private dataCleanup: (() => void) | null = null;

  /**
   * @param mode 'associated' 复用用户终端 PTY，'independent' 创建独立 PTY
   * @param existingPtyId 关联模式下使用的已有 ptyId
   */
  constructor(
    private mode: 'associated' | 'independent' = 'independent',
    private existingPtyId?: string,
  ) {
    super();
  }

  // ── BaseShellExecutor 抽象方法实现 ──

  protected async setupShell(): Promise<void> {
    if (this.mode === 'associated' && this.existingPtyId) {
      // 关联模式：复用已有 PTY（不推荐，会干扰用户输入）
      this.ptyId = this.existingPtyId;
    } else {
      // 独立模式：创建新 PTY 专用于 AI
      const result = await window.electron.localTerminal.create({
        cols: 200,
        rows: 50,
      });
      this.ptyId = result.ptyId;
    }
  }

  protected writeRaw(data: string): void {
    window.electron.localTerminal.write(this.ptyId, data);
  }

  protected onShellDataSetup(callback: (data: string) => void): void {
    this.dataCleanup = window.electron.localTerminal.onData((ptyId, data) => {
      if (ptyId === this.ptyId) {
        callback(data);
      }
    });
  }

  // ── 生命周期 ──

  async initialize(): Promise<void> {
    await this.setupShell();
    this.onShellDataSetup((data) => this.handleShellData(data));
    // 等待 shell 就绪（首次 prompt）
    await this.waitForReady();
  }

  async cleanup(): Promise<void> {
    if (this.dataCleanup) {
      this.dataCleanup();
      this.dataCleanup = null;
    }
    // 独立模式下销毁 PTY
    if (this.mode === 'independent' && this.ptyId) {
      await window.electron.localTerminal.destroy(this.ptyId);
    }
    this.ptyId = '';
  }

  isReady(): boolean {
    return !!this.ptyId;
  }
}
```

```typescript
import { ISystemMonitorService } from './ISystemMonitorService';
import { SystemMetrics } from '../../../../types';

/**
 * 本地系统监控服务
 *
 * 通过 Main 进程 IPC 采集本地系统指标。
 */
export class LocalSystemMonitorService implements ISystemMonitorService {
  private callback: ((metrics: SystemMetrics) => void) | null = null;
  private dataCleanup: (() => void) | null = null;

  onMetrics(callback: (metrics: SystemMetrics) => void): void {
    this.callback = callback;
  }

  start(intervalMs: number = 3000): void {
    // 注册 IPC 数据监听
    this.dataCleanup = window.electron.localMonitor.onData((metrics) => {
      if (this.callback) {
        this.callback(metrics);
      }
    });
    // 通知 Main 进程启动采集
    window.electron.localMonitor.start(intervalMs);
  }

  stop(): void {
    window.electron.localMonitor.stop();
    if (this.dataCleanup) {
      this.dataCleanup();
      this.dataCleanup = null;
    }
  }
}
```

### 6.4 `src/main/local-monitor-service.ts` — Main 进程采集服务

```typescript
import * as os from 'os';
import { exec } from 'child_process';

/**
 * 本地系统监控采集服务（Main 进程）
 *
 * 单例模式，多个 Renderer Tab 共享同一个采集实例。
 */
export class LocalMonitorService {
  private timer: NodeJS.Timeout | null = null;
  private subscribers = new Set<Electron.WebContents>();
  private prevCpuInfo: os.CpuInfo[] | null = null;
  private prevNetBytes: { rx: number; tx: number; time: number } | null = null;

  start(webContents: Electron.WebContents, intervalMs: number): void {
    this.subscribers.add(webContents);
    // 首个订阅者时启动定时器
    if (!this.timer) {
      this.timer = setInterval(() => this.collect(), intervalMs);
      this.collect(); // 立即采集一次
    }
  }

  stop(webContents: Electron.WebContents): void {
    this.subscribers.delete(webContents);
    // 最后一个订阅者离开时停止
    if (this.subscribers.size === 0 && this.timer) {
      clearInterval(this.timer);
      this.timer = null;
    }
  }

  private async collect(): Promise<void> {
    // 采集指标：
    // - CPU: os.cpus() 两次采样差值
    // - 内存: os.freemem() / os.totalmem()
    // - 磁盘: exec('df -h')
    // - 进程: exec('ps aux --sort=-%cpu | head -20')
    // - 网络: exec('netstat -ib') 或 /proc/net/dev
    // - Uptime: os.uptime()
    //
    // 组装为 SystemMetrics 对象
    // 广播给所有 subscribers: webContents.send('local-monitor-data', metrics)
  }
}

export const localMonitorService = new LocalMonitorService();
```

## 9. 实现优先级

| 优先级 | 模块 | 依赖 |
|--------|------|------|
| P0-a | ConnectionInfo 扩展（types.ts） | 无 |
| P0-b | IHostConnection 接口 + Factory | 无 |
| P0-c | SSHHostConnection + LocalHostConnection | 依赖 P0-b + 已有 backends |
| P0-d | TerminalView 重构为使用 IHostConnection | 依赖 P0-a + P0-c |
| P1-a | LocalShellExecutor | 依赖 P0 + 已有 BaseShellExecutor |
| P1-b | useAIAgent 从 IHostConnection 获取 executor | 依赖 P0-d + P1-a |
| P2-a | IFileBrowserService 接口 | 无 |
| P2-b | local-fs-service.ts (Main) + Preload IPC | 无 |
| P2-c | LocalFileBrowserService | 依赖 P2-a + P2-b |
| P2-d | FileBrowserServiceFactory + Panel 适配 | 依赖 P0-a + P2-c |
| P3-a | ISystemMonitorService 接口 | 无 |
| P3-b | local-monitor-service.ts (Main) + Preload IPC | 无 |
| P3-c | LocalSystemMonitorService | 依赖 P3-a + P3-b |
| P3-d | SystemMonitorServiceFactory + 插件适配 | 依赖 P0-a + P3-c |

**并行可能**：P0 完成后，P1（AI Agent）、P2（文件浏览器）、P3（系统监控）三条线可并行开发。P0 是所有子系统的前置基础。
