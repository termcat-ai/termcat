```typescript
/**
 * SSH 终端后端
 *
 * 封装现有 SSH IPC 调用，实现 ITerminalBackend 接口。
 * 不改动 ssh-service.ts，仅做调用层封装。
 */

import { ITerminalBackend } from './ITerminalBackend';
import {
  TerminalConnectOptions,
  SSHConnectOptions,
  TerminalDataCallback,
  TerminalCloseCallback,
} from '@/types/terminal';
import { logger, LOG_MODULE } from '@/utils/logger';

const log = logger.withFields({ module: LOG_MODULE.TERMINAL });

export class SSHTerminalBackend implements ITerminalBackend {
  readonly type = 'ssh' as const;

  private _id: string = '';
  private _isConnected: boolean = false;
  private _dataCallbacks: TerminalDataCallback[] = [];
  private _closeCallbacks: TerminalCloseCallback[] = [];
  private _cleanupFns: (() => void)[] = [];
  private _encoding?: string;

  get id(): string { return this._id; }
  get isConnected(): boolean { return this._isConnected; }

  constructor(private connectionId: string, encoding?: string) {
    this._id = connectionId;
    this._encoding = encoding;
  }

  async connect(options: TerminalConnectOptions): Promise<void> {
    if (!window.electron) {
      throw new Error('Electron API not available');
    }

    log.info('ssh-backend.connecting', 'SSHTerminalBackend connecting', {
      connection_id: this.connectionId,
    });

    // 注册数据监听（必须在 createShell 之前，确保不丢失 MOTD）
    const unsubData = window.electron.onShellData((connId, data) => {
      if (connId === this.connectionId) {
        for (const cb of this._dataCallbacks) {
          cb(data);
        }
      }
    });
    this._cleanupFns.push(unsubData);

    const unsubClose = window.electron.onShellClose((connId) => {
      if (connId === this.connectionId) {
        this._isConnected = false;
        for (const cb of this._closeCallbacks) {
          cb();
        }
      }
    });
    this._cleanupFns.push(unsubClose);

    // 创建 Shell
    await window.electron.sshCreateShell(this.connectionId, this._encoding);
    this._isConnected = true;

    // 通知后端当前终端大小
    await window.electron.sshShellResize(this.connectionId, options.cols, options.rows);

    log.info('ssh-backend.connected', 'SSHTerminalBackend connected', {
      connection_id: this.connectionId,
    });
  }

  async disconnect(): Promise<void> {
    // SSH 断连由上层 sshService 管理，这里不主动调 sshDisconnect
    // 因为一个 SSH 连接可能关联多个 shell（如 AI shell）
    this._isConnected = false;
    log.info('ssh-backend.disconnected', 'SSHTerminalBackend disconnected', {
      connection_id: this.connectionId,
    });
  }

  write(data: string): void {
    if (!window.electron) return;
    window.electron.sshShellWrite(this.connectionId, data)
      .then((result) => {
        if (!result.success) {
          log.error('ssh-backend.write_failed', 'Failed to write', {
            error: 3001, connection_id: this.connectionId,
          });
        }
      })
      .catch((error) => {
        log.error('ssh-backend.write_error', 'Error writing', {
          error: 3002, details: error instanceof Error ? error.message : 'Unknown',
          connection_id: this.connectionId,
        });
      });
  }

  resize(cols: number, rows: number): void {
    if (!window.electron) return;
    window.electron.sshShellResize(this.connectionId, cols, rows);
  }

  onData(callback: TerminalDataCallback): void {
    this._dataCallbacks.push(callback);
  }

  onClose(callback: TerminalCloseCallback): void {
    this._closeCallbacks.push(callback);
  }

  dispose(): void {
    for (const cleanup of this._cleanupFns) {
      cleanup();
    }
    this._cleanupFns = [];
    this._dataCallbacks = [];
    this._closeCallbacks = [];
    if (this._isConnected) {
      this._isConnected = false;
    }
  }
}
```

```typescript
/**
 * 本地 PTY 服务
 *
 * Main 进程中管理本地终端进程，与 ssh-service.ts 平行独立。
 * 使用 node-pty 创建本地 Shell 进程。
 */

import * as pty from 'node-pty';
import * as os from 'os';
import * as fs from 'fs';
import * as path from 'path';
import { logger, LOG_MODULE } from '../utils/logger';

const log = logger.withFields({ module: LOG_MODULE.TERMINAL });

export interface ShellInfo {
  name: string;
  path: string;
  args?: string[];
}

interface PtyInstance {
  id: string;
  process: pty.IPty;
  shell: string;
  cwd: string;
  webContents: Electron.WebContents;
  createdAt: number;
}

export class LocalPtyService {
  private instances = new Map<string, PtyInstance>();

  /**
   * 检测当前平台可用的 Shell 列表
   */
  async detectShells(): Promise<ShellInfo[]> {
    if (process.platform === 'win32') {
      return this.detectWindowsShells();
    }
    return this.detectUnixShells();
  }

  /**
   * 获取当前平台的默认 Shell
   */
  getDefaultShell(): ShellInfo {
    if (process.platform === 'win32') {
      // 优先 pwsh，然后 powershell
      const pwsh = this.findExecutable('pwsh.exe');
      if (pwsh) return { name: 'PowerShell 7', path: pwsh };
      return { name: 'PowerShell', path: 'powershell.exe' };
    }

    const shell = process.env.SHELL || '/bin/sh';
    return { name: path.basename(shell), path: shell };
  }

  /**
   * 创建本地 PTY 进程
   */
  create(options: {
    shell?: string;
    args?: string[];
    cwd?: string;
    env?: Record<string, string>;
    cols: number;
    rows: number;
    webContents: Electron.WebContents;
  }): string {
    const ptyId = `pty-${Date.now()}-${Math.random().toString(36).slice(2, 8)}`;

    const defaultShell = this.getDefaultShell();
    const shell = options.shell || defaultShell.path;
    const args = options.args || defaultShell.args || [];
    const cwd = options.cwd || os.homedir();

    log.info('pty.creating', 'Creating local PTY', {
      pty_id: ptyId, shell, cwd,
    });

    const ptyProcess = pty.spawn(shell, args, {
      name: 'xterm-256color',
      cols: options.cols,
      rows: options.rows,
      cwd,
      env: { ...process.env, ...options.env } as Record<string, string>,
    });

    // 数据输出 → IPC event → Renderer
    ptyProcess.onData((data) => {
      if (!options.webContents.isDestroyed()) {
        options.webContents.send('local-pty-data', ptyId, data);
      }
    });

    // 进程退出
    ptyProcess.onExit(({ exitCode, signal }) => {
      log.info('pty.exited', 'Local PTY exited', {
        pty_id: ptyId, exit_code: exitCode, signal,
      });
      if (!options.webContents.isDestroyed()) {
        options.webContents.send('local-pty-close', ptyId, exitCode);
      }
      this.instances.delete(ptyId);
    });

    this.instances.set(ptyId, {
      id: ptyId,
      process: ptyProcess,
      shell,
      cwd,
      webContents: options.webContents,
      createdAt: Date.now(),
    });

    log.info('pty.created', 'Local PTY created', { pty_id: ptyId });
    return ptyId;
  }

  /**
   * 写入数据
   */
  write(ptyId: string, data: string): boolean {
    const instance = this.instances.get(ptyId);
    if (!instance) return false;
    instance.process.write(data);
    return true;
  }

  /**
   * 调整大小
   */
  resize(ptyId: string, cols: number, rows: number): boolean {
    const instance = this.instances.get(ptyId);
    if (!instance) return false;
    instance.process.resize(cols, rows);
    return true;
  }

  /**
   * 销毁 PTY
   */
  destroy(ptyId: string): void {
    const instance = this.instances.get(ptyId);
    if (!instance) return;
    log.info('pty.destroying', 'Destroying local PTY', { pty_id: ptyId });
    instance.process.kill();
    this.instances.delete(ptyId);
  }

  /**
   * 销毁所有 PTY（应用退出时）
   */
  destroyAll(): void {
    log.info('pty.destroying_all', 'Destroying all local PTY instances', {
      count: this.instances.size,
    });
    for (const [id] of this.instances) {
      this.destroy(id);
    }
  }

  /**
   * 检查 PTY 是否存在
   */
  exists(ptyId: string): boolean {
    return this.instances.has(ptyId);
  }

  // ── Private ──

  private async detectUnixShells(): Promise<ShellInfo[]> {
    const shells: ShellInfo[] = [];
    try {
      const content = await fs.promises.readFile('/etc/shells', 'utf-8');
      const lines = content.split('\n')
        .map(l => l.trim())
        .filter(l => l && !l.startsWith('#'));

      for (const shellPath of lines) {
        try {
          await fs.promises.access(shellPath, fs.constants.X_OK);
          shells.push({
            name: path.basename(shellPath),
            path: shellPath,
          });
        } catch {
          // Shell 不存在或不可执行，跳过
        }
      }
    } catch {
      // /etc/shells 不可读，返回默认
      const fallback = process.env.SHELL || '/bin/sh';
      shells.push({ name: path.basename(fallback), path: fallback });
    }
    return shells;
  }

  private async detectWindowsShells(): Promise<ShellInfo[]> {
    const shells: ShellInfo[] = [];
    const candidates: Array<{ name: string; paths: string[]; args?: string[] }> = [
      { name: 'PowerShell 7', paths: ['pwsh.exe'] },
      { name: 'PowerShell', paths: ['powershell.exe'] },
      { name: 'CMD', paths: ['cmd.exe'] },
      { name: 'WSL', paths: ['wsl.exe'] },
      {
        name: 'Git Bash',
        paths: [
          'C:\\Program Files\\Git\\bin\\bash.exe',
          'C:\\Program Files (x86)\\Git\\bin\\bash.exe',
        ],
        args: ['--login', '-i'],
      },
    ];

    for (const candidate of candidates) {
      const found = this.findExecutableFromPaths(candidate.paths);
      if (found) {
        shells.push({
          name: candidate.name,
          path: found,
          args: candidate.args,
        });
      }
    }
    return shells;
  }

  private findExecutable(name: string): string | null {
    // 尝试在 PATH 中找到可执行文件
    const pathDirs = (process.env.PATH || '').split(path.delimiter);
    for (const dir of pathDirs) {
      const fullPath = path.join(dir, name);
      try {
        fs.accessSync(fullPath, fs.constants.X_OK);
        return fullPath;
      } catch {
        continue;
      }
    }
    return null;
  }

  private findExecutableFromPaths(paths: string[]): string | null {
    for (const p of paths) {
      // 如果是绝对路径，直接检查
      if (path.isAbsolute(p)) {
        try {
          fs.accessSync(p, fs.constants.X_OK);
          return p;
        } catch {
          continue;
        }
      }
      // 否则在 PATH 中查找
      const found = this.findExecutable(p);
      if (found) return found;
    }
    return null;
  }
}

export const localPtyService = new LocalPtyService();
```

## Task 8: LocalTerminalBackend 实现

**Files:**
- Create: `src/services/terminal-backends/LocalTerminalBackend.ts`

- [ ] **Step 1: 创建 `LocalTerminalBackend.ts`**

```typescript
/**
 * 本地终端后端
 *
 * 封装 Local PTY IPC 调用，实现 ITerminalBackend 接口。
 */

import { ITerminalBackend } from './ITerminalBackend';
import {
  TerminalConnectOptions,
  LocalConnectOptions,
  TerminalDataCallback,
  TerminalCloseCallback,
} from '@/types/terminal';
import { logger, LOG_MODULE } from '@/utils/logger';

const log = logger.withFields({ module: LOG_MODULE.TERMINAL });

export class LocalTerminalBackend implements ITerminalBackend {
  readonly type = 'local' as const;

  private _id: string = '';
  private _isConnected: boolean = false;
  private _dataCallbacks: TerminalDataCallback[] = [];
  private _closeCallbacks: TerminalCloseCallback[] = [];
  private _cleanupFns: (() => void)[] = [];
  private _localOptions: Partial<LocalConnectOptions>;

  get id(): string { return this._id; }
  get isConnected(): boolean { return this._isConnected; }

  constructor(options?: Partial<LocalConnectOptions>) {
    this._localOptions = options || {};
  }

  async connect(options: TerminalConnectOptions): Promise<void> {
    if (!window.electron?.localTerminal) {
      throw new Error('Local terminal API not available');
    }

    log.info('local-backend.connecting', 'LocalTerminalBackend connecting', {
      shell: this._localOptions.shell,
      cwd: this._localOptions.cwd,
    });

    // 注册数据监听（在 create 之前，确保不丢失首批输出）
    const unsubData = window.electron.localTerminal.onData((ptyId, data) => {
      if (ptyId === this._id) {
        for (const cb of this._dataCallbacks) {
          cb(data);
        }
      }
    });
    this._cleanupFns.push(unsubData);

    const unsubClose = window.electron.localTerminal.onClose((ptyId) => {
      if (ptyId === this._id) {
        this._isConnected = false;
        for (const cb of this._closeCallbacks) {
          cb();
        }
      }
    });
    this._cleanupFns.push(unsubClose);

    // 创建 PTY
    const result = await window.electron.localTerminal.create({
      shell: this._localOptions.shell,
      args: this._localOptions.args,
      cwd: this._localOptions.cwd,
      env: this._localOptions.env,
      cols: options.cols,
      rows: options.rows,
    });

    this._id = result.ptyId;
    this._isConnected = true;

    log.info('local-backend.connected', 'LocalTerminalBackend connected', {
      pty_id: this._id,
    });
  }

  async disconnect(): Promise<void> {
    if (!window.electron?.localTerminal) return;
    if (this._id) {
      await window.electron.localTerminal.destroy(this._id);
    }
    this._isConnected = false;
    log.info('local-backend.disconnected', 'LocalTerminalBackend disconnected', {
      pty_id: this._id,
    });
  }

  write(data: string): void {
    if (!window.electron?.localTerminal || !this._id) return;
    window.electron.localTerminal.write(this._id, data);
  }

  resize(cols: number, rows: number): void {
    if (!window.electron?.localTerminal || !this._id) return;
    window.electron.localTerminal.resize(this._id, cols, rows);
  }

  onData(callback: TerminalDataCallback): void {
    this._dataCallbacks.push(callback);
  }

  onClose(callback: TerminalCloseCallback): void {
    this._closeCallbacks.push(callback);
  }

  dispose(): void {
    for (const cleanup of this._cleanupFns) {
      cleanup();
    }
    this._cleanupFns = [];
    this._dataCallbacks = [];
    this._closeCallbacks = [];
    if (this._isConnected) {
      this.disconnect();
    }
  }
}
```

 * 根据 Session 的 connectionType 创建对应的 Backend 实例。

import { ITerminalBackend } from './ITerminalBackend';
import { SSHTerminalBackend } from './SSHTerminalBackend';
import { LocalTerminalBackend } from './LocalTerminalBackend';
import { Session } from '@/types';

export class TerminalBackendFactory {
  /**
   * 根据 session 信息创建终端后端
   *
   * @param session - 终端会话
   * @param connectionId - SSH 连接 ID（SSH 模式必须）
   */
  static create(session: Session, connectionId?: string): ITerminalBackend {
    if (session.host?.connectionType === 'local') {

        env: session.host.localConfig?.env,
      });

      throw new Error('connectionId is required for SSH terminal backend');
    }
    return new SSHTerminalBackend(
      connectionId,
      session.host?.terminal?.encoding,
    );
  }
}
